(function () { 'use strict'; function RemovePlugin(pluginToRemoveUrl) { var pluginsArray = Lampa.Storage.get('plugins') || []; var updatedPlugins = pluginsArray.filter(function (plugin) { return plugin.url !== pluginToRemoveUrl; }); if (updatedPlugins.length > 0) { Lampa.Storage.set('plugins', updatedPlugins); Lampa.Storage.set('needReboot', true); Lampa.Noty.show("Установка завершена, установщик удалён\nТребуется перезагрузка"); Lampa.Storage.set('needRebootSettingExit', true); console.log('RemovePlugin()', 'Плагины с URL', pluginToRemoveUrl, 'удалены'); } else { console.log('RemovePlugin()', 'Плагин с URL', pluginToRemoveUrl, 'не найден'); } window.location.reload(); // Перезагрузка страницы } function getTextFromURL(url, callback) { var xhr = new XMLHttpRequest(); xhr.onreadystatechange = function () { if (xhr.readyState === XMLHttpRequest.DONE) { if (xhr.status === 200) { callback(null, xhr.responseText); } else { callback('Ошибка загрузки: ' + xhr.status); } } }; xhr.open('GET', url, true); xhr.send(); console.log('install', 'Запрос текста по URL:', url); } function processPluginsText(err, text) { if (err) { console.error('install', err); return; } var blocks = text.split(/\n\s*\n/); var pluginsArray = []; var pluginsToDisable = []; blocks.forEach(function (block) { if (block.startsWith('[Тест]') || block.startsWith('[Установщик]')) { console.log('install', 'Игнорируем блок:', block); return; } var lines = block.split('\n'); lines.forEach(function (line) { if (line.trim() !== '' && line.includes('-')) { var parts = line.split(' - '); var id = parts[0].trim(); var url = parts[1].trim(); var authorURL = parts[2].trim(); var name = parts[3].trim(); var author = '@lampa'; if (authorURL.trim() !== '') { var authorMatch = authorURL.match(/\/\/([^\.]+)\./); if (authorMatch) { author = '@' + authorMatch[1]; } else if (!authorURL.includes('://')) { author = authorURL.split(' ')[0]; } } var status = 1; var plugin = { author: author, url: url, name: name, status: status, description: name }; if (id !== 'NotMobile' && id !== 'Interface') { plugin.id = id; pluginsArray.push(plugin); console.log('install', 'Плагин установлен:', name); } else { // Добавляем плагин в список для отключения в конце pluginsToDisable.push(plugin); console.log('install', 'Плагин добавлен в список для отключения:', name); } } }); }); // Добавляем плагины "NotMobile" и "Interface" в самом конце pluginsToDisable.forEach(function (plugin) { if (plugin.id === 'Interface') { // Перемещаем "Interface" в самый конец списка pluginsArray.push(plugin); console.log('install', 'Плагин "Interface" установлен в самую последнюю очередь:', plugin.name); } else { pluginsArray.push(plugin); console.log('install', 'Плагин установлен в самом конце:', plugin.name); } }); var selfPlugin = { author: '@lat5', url: 'https://lat5.github.io/pl/Install', name: 'Automated Plugin Installer', status: 0, description: 'Автоматическая установка всех плагинов из\nhttps://lat5.github.io/pl/\nКроме блока \'Тест\'' }; var selfPluginsArray = Lampa.Storage.get('plugins') || []; selfPluginsArray.push(selfPlugin); Lampa.Storage.set('plugins', selfPluginsArray); console.log('install', 'Сама программа успешно изменена:', selfPlugin); RemovePlugin('https://lat5.github.io/pl/Install'); } console.log('install', 'Запуск программы...'); getTextFromURL('https://raw.githubusercontent.com/lat5/pl/main/plugins_list.txt', processPluginsText); })();
