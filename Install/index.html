(function () { 'use strict'; // Функция для удаления плагина по его URL function RemovePlugin(pluginToRemoveUrl) { var pluginsArray = Lampa.Storage.get('plugins') || []; var updatedPlugins = pluginsArray.filter(function (plugin) { return plugin.url !== pluginToRemoveUrl; }); // Проверяем, были ли найдены плагины для удаления if (updatedPlugins.length > 0) { Lampa.Storage.set('plugins', updatedPlugins); console.log('install', 'Плагины с URL', pluginToRemoveUrl, 'удалены'); } else { console.log('install', 'Плагин с URL', pluginToRemoveUrl, 'не найден'); } } // Функция для получения текста с URL function getTextFromURL(url, callback) { var xhr = new XMLHttpRequest(); xhr.onreadystatechange = function () { if (xhr.readyState === XMLHttpRequest.DONE) { if (xhr.status === 200) { callback(null, xhr.responseText); } else { callback('Ошибка загрузки: ' + xhr.status); } } }; xhr.open('GET', url, true); xhr.send(); console.log('install', 'Запрос текста по URL:', url); } // Функция для обработки текста и установки плагинов function processPluginsText(err, text) { if (err) { console.error('install', err); return; } var blocks = text.split(/\n\s*\n/); // Разделение текста на блоки blocks.forEach(function (block) { // Игнорируем блоки, начинающиеся с [Тест] и [Install] if (block.startsWith('[Тест]') || block.startsWith('[Install]')) { console.log('install', 'Игнорируем блок:', block); return; } var lines = block.split('\n'); lines.forEach(function (line) { if (line.trim() !== '' && line.includes('-')) { var parts = line.split(' - '); var id = parts[0].trim(); var url = parts[1].trim(); var authorURL = parts[2].trim(); var name = parts[3].trim(); // Извлекаем имя автора var author = '@lampa'; // По умолчанию if (authorURL.trim() !== '') { var authorMatch = authorURL.match(/\/\/([^\.]+)\./); if (authorMatch) { author = '@' + authorMatch[1]; } else if (!authorURL.includes('://')) { // Если нет '://' в URL автора, используем первое слово author = authorURL.split(' ')[0]; } } // Устанавливаем плагин var status = 1; if (id === 'NotMobile' || id === 'Interface') { status = 0; } var pluginsArray = Lampa.Storage.get('plugins') || []; var plugin = { author: author, url: url, name: name, status: status, description: name }; // Проверяем, передавать ли ID if (id !== 'NotMobile' && id !== 'Interface') { plugin.id = id; } pluginsArray.push(plugin); Lampa.Storage.set('plugins', pluginsArray); console.log('install', 'Плагин установлен:', name); } }); }); // После завершения выполнения программы меняем её саму var selfPlugin = { author: '@lat5', url: 'https://lat5.github.io/pl/Install', name: 'Automated Plugin Installer', status: 0, description: 'Автоматическая установка всех плагинов из\nhttps://lat5.github.io/pl/\nКроме блока \'Тест\'' }; var selfPluginsArray = Lampa.Storage.get('plugins') || []; selfPluginsArray.push(selfPlugin); Lampa.Storage.set('plugins', selfPluginsArray); console.log('install', 'Сама программа успешно изменена:', selfPlugin); // Удалить плагин-установщик RemovePlugin('https://lat5.github.io/pl/Install'); } console.log('install', 'Запуск программы...'); // Загружаем текст со страницы и обрабатываем его getTextFromURL('https://raw.githubusercontent.com/lat5/pl/main/plugins_list.txt', processPluginsText); })();
