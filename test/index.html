(function () { 'use strict'; function displayStorageVariables() { // Получаем все ключи из Lampa.Storage var keys = Object.keys(Lampa.Storage) console.log("test", "Поиск"); console.log("test", keys); // Перебираем ключи keys.forEach(function(key) { // Получаем значение для каждого ключа var value = Lampa.Storage.get(key); // Проверяем, не пустое ли значение if (value !== null && value !== undefined && value !== '') { // Если значение не пустое, выводим его в консоль с префиксом "test" console.log("test", key + ":", value); } }); } // Вызываем функцию для отображения переменных в консоли displayStorageVariables(); })();
